	myFunction : function(){
	console.log(this == myObj);
	}
}
myObj.myFunction()  // prints true


2.
var myObj = {
	myFunction : function(){
	console.log(this == myObj);
	}
}
myObj.myFunction()  // prints false

explanation : if its is called in the form obj.function "this" equals "obj" else "this" equals "global"
